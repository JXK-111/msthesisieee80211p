1function [y,NoiseSigma] = add_awgn_noise(x,snrdB)
 2    %y=awgn_noise(x,SNR) adds AWGN noise vector to signal 'x' to generate a
 3    %resulting signal vector y of specified SNR in dB
 4    %rng('default');%set the random generator seed to default (for comparison only)
 5    L=length(x);
 6    SNR = 10^(snrdB/10); %SNR to linear scale
 7    Esym=sum(abs(x).^2)/(L); %Calculate actual symbol energy
 8    N0=Esym/SNR; %Find the noise spectral density
 9    if(isreal(x))
10        NoiseSigma = sqrt(N0);%Standard deviation for AWGN Noise when x is real
11        n = NoiseSigma*randn(1,L);%computed noise
12    else
13        NoiseSigma=sqrt(N0/2);%Standard deviation for AWGN Noise when x is complex
14        n = NoiseSigma*(randn(1,L)+1i*randn(1,L));%computed noise
15    end    
16    y = x + n; %received signal    
17    
18end
